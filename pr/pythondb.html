1. Conexión a MySQL desde Python

import mysql.connector

# Establecer conexión
conn = mysql.connector.connect(
    host="localhost",
    user="root",
    password="tu_contraseña",
    database="tu_base_de_datos"
)

if conn.is_connected():
    print("Conectado a MySQL")
else:
    print("No se pudo conectar")

2. Insertar Datos en la Base de Datos

cursor = conn.cursor()

# Insertar datos
query = "INSERT INTO usuarios (nombre, edad) VALUES (%s, %s)"
valores = ("Juan", 30)
cursor.execute(query, valores)
conn.commit()

print(cursor.rowcount, "registro(s) insertado(s)")

3. Consultar Datos

# Consultar datos
cursor.execute("SELECT * FROM usuarios")
resultados = cursor.fetchall()

for fila in resultados:
    print(fila)

4. Actualizar Datos

# Actualizar datos
query = "UPDATE usuarios SET edad = %s WHERE nombre = %s"
valores = (31, "Juan")
cursor.execute(query, valores)
conn.commit()

print(cursor.rowcount, "registro(s) actualizado(s)")

5. Eliminar Datos

# Eliminar datos
query = "DELETE FROM usuarios WHERE nombre = %s"
valores = ("Juan",)
cursor.execute(query, valores)
conn.commit()

print(cursor.rowcount, "registro(s) eliminado(s)")
